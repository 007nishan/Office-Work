    function getAddressId() {
        const allDivs = Array.from(document.querySelectorAll('div'));
        const aidDiv = allDivs.find(div => div.textContent.includes('Address ID:'));
        if (aidDiv) {
            const match = aidDiv.textContent.match(/Address ID:\s*([A-Z0-9]+)/);
            return match ? String(match) : null;
        }
        return null;
    }

    function extractPDLLocation(value) {
        if (!value) return value;
        const cleanedValue = value.replace(/^["']|["']$/g, '');
        try {
            const match = cleanedValue.match(/"SafePlaceLocation_":"([^"]+)"/i);
            if (match) return match[1];

            const jsonMatch = cleanedValue.match(/:\s*"([^"]+)"\s*\}/);
            if (jsonMatch) return jsonMatch[1];
            
            const anyMatch = cleanedValue.match(/:\s*"([^"]+)"/g);
            if (anyMatch && anyMatch.length > 0) {
                 const lastMatch = anyMatch[anyMatch.length - 1];
                 const val = lastMatch.match(/"([^"]+)"$/);
                 if (val) return val[1];
            }

            return cleanedValue;
        } catch (e) {
            return cleanedValue;
        }
    }

            function determineCDP(dhData, pdlData) {
                const allEntries = [
                    ...dhData.map(d => ({...d, Type: 'Delivery Hint'})),
                    ...pdlData.map(d => ({...d, Type: 'Preferred Delivery Location'}))
                ];

                const customerEntries = allEntries.filter(e => e.Source === 'CUSTOMER');
                const llmEntries = allEntries.filter(e => e.Source.includes('LLM_DERIVED'));
                const otherEntries = allEntries.filter(e => e.Source !== 'CUSTOMER' && !e.Source.includes('LLM_DERIVED'));

                let cdp = null;
                let reasoning = '';

                if (customerEntries.length > 0) {
                    const dhCustomer = dhData.filter(d => d.Source === 'CUSTOMER');
                    const pdlCustomer = pdlData.filter(d => d.Source === 'CUSTOMER');

                    if (dhCustomer.length > 0 && pdlCustomer.length > 0) {
                        let hasBothCorrect = false;
                        for (const dh of dhCustomer) {
                            for (const pdl of pdlCustomer) {
                                if (dh.Timestamp === pdl.Timestamp) {
                                    hasBothCorrect = true;
                                    cdp = {
                                        Type: 'DH_PDL Both Correct',
                                        Value: 'Both DH and PDL valid locations',
                                        Source: 'CUSTOMER',
                                        Timestamp: dh.Timestamp
                                    };
                                    reasoning = 'DH and PDL have identical Customer source and timestamp (' + dh.Timestamp + '). No ground for comparison exists - cannot determine priority between DH and PDL.';
                                    break;
                                }
                            }
                            if (hasBothCorrect) break;
                        }

                        if (!hasBothCorrect) {
                            customerEntries.sort((a, b) => new Date(b.Timestamp) - new Date(a.Timestamp));
                            cdp = customerEntries.shift();
                            const selectedType = cdp.Type === 'Delivery Hint' ? 'DH' : 'PDL';
                            reasoning = 'Both DH and PDL have Customer source. ' + selectedType + ' timestamp (' + cdp.Timestamp + ') is more recent. Selected ' + selectedType + ' as CDP.';
                        }
                    } else if (dhCustomer.length > 0) {
                        customerEntries.sort((a, b) => new Date(b.Timestamp) - new Date(a.Timestamp));
                        cdp = customerEntries.shift();
                        reasoning = 'Only DH has Customer source. Selected DH as CDP.';
                    } else if (pdlCustomer.length > 0) {
                        customerEntries.sort((a, b) => new Date(b.Timestamp) - new Date(a.Timestamp));
                        cdp = customerEntries.shift();
                        reasoning = 'Only PDL has Customer source. Selected PDL as CDP.';
                    }
                } else if (llmEntries.length > 0) {
                    llmEntries.sort((a, b) => new Date(b.Timestamp) - new Date(a.Timestamp));
                    cdp = llmEntries.shift();
                    reasoning = 'No Customer source. LLM_DERIVED has second priority. Selected most recent LLM entry as CDP.';
                } else if (otherEntries.length > 0) {
                    otherEntries.sort((a, b) => new Date(b.Timestamp) - new Date(a.Timestamp));
                    cdp = otherEntries.shift();
                    reasoning = 'No Customer or LLM source. Selected most recent from other sources as CDP.';
                }

                return { cdp, reasoning };
            }

            function detectDHPDLBothCorrect(dhData, pdlData) {
                const dhBySource = {};
                const pdlBySource = {};

                dhData.forEach(d => {
                    if (!dhBySource[d.Source]) dhBySource[d.Source] = [];
                    dhBySource[d.Source].push(d);
                });

                pdlData.forEach(d => {
                    if (!pdlBySource[d.Source]) pdlBySource[d.Source] = [];
                    pdlBySource[d.Source].push(d);
                });

                for (const source in dhBySource) {
                    if (pdlBySource[source]) {
                        for (const dh of dhBySource[source]) {
                            for (const pdl of pdlBySource[source]) {
                                if (dh.Timestamp === pdl.Timestamp) {
                                    return true;
                                }
                            }
                        }
                    }
                }
                return false;
            }

            const { cdp, reasoning } = determineCDP(dhData, pdlData);
            const hasBothCorrect = detectDHPDLBothCorrect(dhData, pdlData);

            isExtracting = false;

            return {
                cdp: cdp,
                reasoning: reasoning,
                dhData: dhData,
                pdlData: pdlData,
                hasBothCorrect: hasBothCorrect,
                extractedAt: new Date().toISOString()
            };

        } catch (error) {
            console.log('Error during extraction: ' + error.message);
            isExtracting = false;
            return null;
        }
    }

    function setupDragging() {
        let startX, startY, startTop, startRight;

        cdpPanel.addEventListener('mousedown', (e) => {
            if (e.target.tagName === 'BUTTON' || e.target.innerHTML === '×') return;
            isDragging = true;
            startX = e.clientX;
            startY = e.clientY;
            startTop = parseInt(cdpPanel.style.top);
            startRight = parseInt(cdpPanel.style.right);
            cdpPanel.style.cursor = 'grabbing';
        });

        document.addEventListener('mousemove', (e) => {
            if (isDragging) {
                const deltaX = e.clientX - startX;
                const deltaY = e.clientY - startY;
                cdpPanel.style.top = (startTop + deltaY) + 'px';
                cdpPanel.style.right = (startRight - deltaX) + 'px';
            }
        });

        document.addEventListener('mouseup', () => {
            if (isDragging) {
                isDragging = false;
                cdpPanel.style.cursor = 'move';
                savePosition(cdpPanel.style.top, cdpPanel.style.right);
            }
        });
    }

    function createAnalyticsEntry(num, data, isWinner = false) {
        const entry = document.createElement('div');
        Object.assign(entry.style, {
            padding: '12px',
            marginBottom: '8px',
            backgroundColor: isWinner ? '#e6f4ea' : '#f7f8f8',
            border: isWinner ? '2px solid #34a853' : 'none',
            borderRadius: '8px',
            fontSize: '14px',
            lineHeight: '1.5',
            color: '#16191f',
            wordWrap: 'break-word',
            overflowWrap: 'break-word'
        });

        let sourceDisplay = data.Source;
        let timestampDisplay = data.Timestamp;

        if (data.Source && data.Source.toUpperCase() === 'CUSTOMER') {
            sourceDisplay = `<span style="color: #d13212; font-weight: bold;">${data.Source}</span>`;
            const tIdx = data.Timestamp ? data.Timestamp.indexOf('T') : -1;
            if (tIdx !== -1) {
                const dp = data.Timestamp.substring(0, tIdx);
                const tp = data.Timestamp.substring(tIdx);
                timestampDisplay = `<span style="color: #d13212; font-weight: bold;">${dp}</span><span style="color: #16191f;">${tp}</span>`;
            } else {
                timestampDisplay = `<span style="color: #d13212; font-weight: bold;">${data.Timestamp}</span>`;
            }
        }

        entry.innerHTML = '<b>' + num + '.</b><br><b>Value:</b> ' + data.Value + '<br><b>Source:</b> ' + sourceDisplay + '<br><b>Timestamp:</b> ' + timestampDisplay;

    function monitorCases() {
        setInterval(async () => {
            const now = Date.now();
            if (now - lastCheckTime >= 500) {
                lastCheckTime = now;
                const currentAid = getAddressId();

                if (currentAid && currentAid !== lastProcessedAddressId && !isExtracting) {
                    console.log('New case detected: ' + currentAid + ' (previous: ' + lastProcessedAddressId + ')');
                    lastProcessedAddressId = currentAid;

                    if (cdpPanel) {
                        while (cdpPanel.childNodes.length > 0) {
                            cdpPanel.removeChild(cdpPanel.lastChild);
                        }
                        const loading = document.createElement('div');
                        loading.textContent = 'Loading...';
                        Object.assign(loading.style, {
                            fontSize: '14px',
                            fontFamily: 'Amazon Ember, Helvetica, Arial, sans-serif',
                            textAlign: 'center',
                            color: '#545b64',
                            marginTop: '24px'
                        });
                        cdpPanel.appendChild(loading);
                    }

                    const data = await extractCDPData();
                    cachedData = data;
                    updateCDPBoardContent(data);
                }
            }
        }, 500);
    }
