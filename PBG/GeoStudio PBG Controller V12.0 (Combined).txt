  async function waitFor(selectorOrFn, timeout = 10000, interval = 200) {
    const start = Date.now();
    while (Date.now() - start < timeout) {
      try {
        const el = (typeof selectorOrFn === 'function')
          ? selectorOrFn()
          : document.querySelector(selectorOrFn);
        if (el) return el;
      } catch (e) {}
      await delay(interval);
    }
    return null;
  }

      clean(a) { return a.replace(/[,\/#!$%\^&\*;:{}=\-_`~()]/g, " ").replace(/\s+/g, " ").trim(); },
      standardizeToken(t) { return ABBREVIATIONS[t.toUpperCase()] || t; },
      compare(unitAddr, custAddr) {
        try {
          if (!unitAddr || !custAddr) return { isMatch: false, differences: { missing: true } };
          const cu = this.clean(unitAddr), cc = this.clean(custAddr);
          const pr = /\b\d{5}(-\d{4})?\b/;
          const up = cu.match(pr), cp = cc.match(pr);
          const diff = { postalCode: null, tokens: null };
          if (up || cp) {
            const ub = up ? up[0].substring(0, 5) : null, cb = cp ? cp[0].substring(0, 5) : null;
            if (ub !== cb) diff.postalCode = { unit: up ? up[0] : "missing", customer: cp ? cp[0] : "missing" };
          }
          const ut = cu.replace(pr, "").trim().split(" ").filter(t => t.length > 0).map(t => this.standardizeToken(t));
          const ct = cc.replace(pr, "").trim().split(" ").filter(t => t.length > 0).map(t => this.standardizeToken(t));
          const uu = ut.filter(t => !ct.includes(t)), uc = ct.filter(t => !ut.includes(t));
          let aptDiff = false;
          const uai = ut.findIndex(t => ["APARTMENT", "APT", "UNIT"].includes(t.toUpperCase()));
          const cai = ct.findIndex(t => ["APARTMENT", "APT", "UNIT"].includes(t.toUpperCase()));
          if (uai !== -1 && cai !== -1 && ut[uai + 1] !== ct[cai + 1]) aptDiff = true;
          if (uu.length > 0 || uc.length > 0 || aptDiff) diff.tokens = { unit: uu.length > 0 ? uu : null, customer: uc.length > 0 ? uc : null, apartmentDifference: aptDiff };
          const has = Object.values(diff).some(d => d !== null);
          return { isMatch: !has, differences: has ? diff : null };
        } catch (e) { return { isMatch: false, differences: { error: true } }; }
      },
    };

        for (const p of CONFIG.QUEUE_PATTERNS.SUGGESTED_GEOCODES) { if (u.includes(p.toUpperCase())) return { type: "SUGGESTED_GEOCODES", pattern: p }; }

    const GeofenceValidator = {
      validate(value, cd) {
        const n = parseFloat(value); if (isNaN(n)) return { valid: false, message: "Invalid value" };
        const isR = (cd.Geofencereduction25m || "").toLowerCase() === "yes";
        const isC = (cd.Geofencecreation || "").toLowerCase() === "yes";
        const min = isR ? CONFIG.GEOFENCE.MIN_REDUCTION : CONFIG.GEOFENCE.MIN_STANDARD;
        if (isC && n === CONFIG.GEOFENCE.CREATION_PLACEHOLDER) return { valid: true, message: "Creation case with -1 value (min: " + min + "m)" };
        if (n < min) return { valid: false, message: "Too low. Min: " + min + "m (Reduction " + (isR ? "on" : "off") + ")" };
        return { valid: true, message: "Valid (min: " + min + "m, Reduction " + (isR ? "on" : "off") + ")" };
      },
    };

    const ClassificationValidator = {
      classify(pdl, dh) {
        pdl = (pdl || "").trim(); dh = (dh || "").trim();
        if (TextUtils.hasLocationPattern(pdl)) return { classification: "ADDRESS LEVEL", needsVerification: false, reason: "PDL contains specific location" };
        if (TextUtils.isEmpty(pdl) && TextUtils.isEmpty(dh)) return { classification: null, needsVerification: true, reason: "No CDP information - verification required for gate/mailroom cases" };
        if (TextUtils.isFrontDoor(pdl) && TextUtils.isEmpty(dh)) return { classification: "UNIT LEVEL", needsVerification: false, reason: "Front door PDL with empty DH" };
        if (TextUtils.isFrontDoor(pdl) && TextUtils.isFrontDoor(dh)) return { classification: "UNIT LEVEL", needsVerification: false, reason: "Both PDL and DH are front door" };
        if (TextUtils.isFrontDoor(pdl) && TextUtils.hasAdditionalText(dh)) return { classification: null, needsVerification: true, reason: "Front door PDL with additional DH text" };
        if (TextUtils.isEmpty(pdl) && !TextUtils.isEmpty(dh)) return { classification: null, needsVerification: true, reason: "Empty PDL with DH text" };
        if (!TextUtils.isEmpty(pdl) && !TextUtils.hasLocationPattern(pdl) && !TextUtils.isFrontDoor(pdl)) return { classification: null, needsVerification: true, reason: "PDL has unclear information" };
        return { classification: "ADDRESS LEVEL", needsVerification: false, reason: "Default to address level" };
      },
    };

        await this.delay(300);
        return box.getAttribute('aria-expanded') === 'true';
      },
      async selectOption(label) {
        for (let attempt = 0; attempt < 3; attempt++) {
          const options = document.querySelectorAll('div[role="option"], li[role="option"]');
          for (const opt of options) {
            if (opt.textContent.trim() === label) {
              opt.dispatchEvent(new MouseEvent('mousedown', { bubbles: true, cancelable: true }));
              await this.delay(30);
              opt.dispatchEvent(new MouseEvent('mouseup', { bubbles: true, cancelable: true }));
              await this.delay(30);
              opt.click(); return true;
            }
          }
          const listboxId = document.querySelector('div[role="combobox"][aria-expanded="true"]')?.getAttribute('aria-controls');
          if (listboxId) {
            const listbox = document.getElementById(listboxId);
            if (listbox) {
              const items = listbox.querySelectorAll('*');
              for (const item of items) {
                if (item.textContent.trim() === label && item.children.length === 0) {
                  item.dispatchEvent(new MouseEvent('mousedown', { bubbles: true, cancelable: true }));
                  await this.delay(30); item.click(); return true;
                }
              }
            }
          }
          await this.delay(200);
        }
        return false;
      },
      async waitForEnabled(box, timeout = 3000) {
        const start = Date.now();
        while (Date.now() - start < timeout) {
          if (box.getAttribute('aria-disabled') !== 'true') return true;
          await this.delay(200);
        }
        return false;
      },
      isAlreadyConfigured() {
        const acc = this.findPastDeliveriesAccordion();
        if (!acc) return false;
        const comboboxes = acc.querySelectorAll('div[role="combobox"][mdn-input-box]');
        let hasCount = false, hasAll = false;
        for (const box of comboboxes) {
          const valueEl = box.querySelector('[mdn-select-value]');
          if (!valueEl) continue;
          const text = valueEl.textContent.trim();
          if (text === 'Count') hasCount = true;
          if (text === 'All') hasAll = true;
        }
        return hasCount && hasAll;
      },
      async openPastDeliveries() {
        if (this.isPastDeliveriesExpanded()) return true;
        const acc = this.findPastDeliveriesAccordion();
        if (!acc) return false;
        const summary = acc.querySelector('.MuiAccordionSummary-root');
        if (!summary) return false;
        const pastBtn = acc.querySelector(DOM_SELECTORS.PAST_DEL_TEXT);
        if (pastBtn) pastBtn.scrollIntoView({ behavior: "smooth", block: "center" });
        await this.delay(200);
        summary.click(); await this.delay(2000);
        return this.isPastDeliveriesExpanded();
      },
      async waitForComboboxes(timeout = 5000) {
        const start = Date.now();
        while (Date.now() - start < timeout) {
          if (this.hasComboboxesInAccordion()) return true;
          await this.delay(300);
        }
        return false;
      },
      async configureAttributeAndFilter() {
        try {
          if (!this.isPastDeliveriesExpanded()) {
            const opened = await this.openPastDeliveries();
            if (!opened) return false;
          }
          await this.waitForComboboxes();
          if (this.isAlreadyConfigured()) return true;
          let attrBox = this.findMeridianComboboxInPastDel('Attribute') || this.findMeridianComboboxInPastDel('Count');
          if (!attrBox) return false;
          const valueEl = attrBox.querySelector('[mdn-select-value]');
          if (!valueEl || valueEl.textContent.trim() !== 'Count') {
            const opened = await this.openMeridianCombobox(attrBox);
            if (!opened) return false;
            await this.delay(300);
            const selected = await this.selectOption('Count');
            if (!selected) return false;
            await this.delay(500);
          }
          let filterBox = this.findMeridianComboboxInPastDel('Filter') ||
            this.findMeridianComboboxInPastDel('Recent') ||
            this.findMeridianComboboxInPastDel('All');
          if (!filterBox) {
            const acc = this.findPastDeliveriesAccordion();
            if (acc) {
              const comboboxes = acc.querySelectorAll('div[role="combobox"][mdn-input-box]');
              const countBox = this.findMeridianComboboxInPastDel('Count');
              for (const box of comboboxes) {
                if (box !== attrBox && box !== countBox) { filterBox = box; break; }
              }
            }
          }
          if (filterBox) {
            const filterValue = filterBox.querySelector('[mdn-select-value]');
            if (filterValue && filterValue.textContent.trim() === 'All') return true;
            const enabled = await this.waitForEnabled(filterBox);
            if (!enabled) return false;
            await this.delay(300);
            const filterOpened = await this.openMeridianCombobox(filterBox);
            if (!filterOpened) return false;
            await this.delay(300);
            await this.selectOption('All');
            await this.delay(300);
          }
          return true;
        } catch (e) { return false; }
      },
      async autoOpenAndSetFilters() {
        if (this.isProcessing) return;
        this.isProcessing = true;
        try {
          await this.delay(1500);
          const success = await this.configureAttributeAndFilter();
          if (!success && this.retryAttempt < 2) {
            this.retryAttempt++;
            this.isProcessing = false;
            await this.delay(2000);
            return await this.autoOpenAndSetFilters();
          }
          this.retryAttempt = 0;
        } catch (e) { this.retryAttempt = 0; }
        this.isProcessing = false;
      },
      async searchTracking(trackingId) {
        if (this.isTrackingSearching || !trackingId) return;
        this.isTrackingSearching = true;
        try {
          this.clearHighlights();
          if (!this.isPastDeliveriesExpanded()) {
            await this.openPastDeliveries();
            await this.waitForComboboxes();
            await this.configureAttributeAndFilter();
            await this.delay(800);
          } else if (!this.isAlreadyConfigured()) {
            await this.configureAttributeAndFilter();
            await this.delay(500);
          }
          const result = this.scanForTracking(trackingId);
          if (result.found) {
            result.row.classList.add('pbg-delivery-highlight');
            result.row.scrollIntoView({ behavior: 'smooth', block: 'center' });
            await this.delay(400);

            const photoBtn = result.row.querySelector(DOM_SELECTORS.PAST_DEL_PHOTO_BTN);
            const photoStatus = photoBtn ? 'opened' : 'no-button';
            if (photoBtn) photoBtn.click();
            CaseDetailsManager.trackingSearchResult = {
              trackingId, searched: true, found: true,
              delivery: { date: result.date, reason: result.reason, defect: result.defect, placeId: result.placeId, photoStatus },
            };
            // Highlight persists until new case loads (cleared in reset())
          } else {
            CaseDetailsManager.trackingSearchResult = { trackingId, searched: true, found: false, delivery: null };
            // Show a visible notification popup for no match
            NotificationManager.show("WARNING", "Tracking Not Found",
              trackingId + "\n\nNo matching delivery found in Past Deliveries.",
              CONFIG.NOTIFICATION_DURATION.CLASSIFIER);
          }
          if (CaseDetailsManager.caseData) CaseDetailsPanel.update(CaseDetailsManager.caseData, false, null);
        } catch (e) {
          CaseDetailsManager.trackingSearchResult = { trackingId, searched: true, found: false, delivery: null };
          NotificationManager.show("ERROR", "Search Failed", "Could not search for tracking ID. Please try again.", CONFIG.NOTIFICATION_DURATION.ERROR);
          if (CaseDetailsManager.caseData) CaseDetailsPanel.update(CaseDetailsManager.caseData, false, null);
        }
        this.isTrackingSearching = false;
      },
      scanForTracking(trackingId) {
        const rows = document.querySelectorAll(DOM_SELECTORS.PAST_DEL_ROW);
        for (const row of rows) {
          const tbaEl = row.querySelector('.css-16947m2 ' + DOM_SELECTORS.PAST_DEL_TBA);
          if (tbaEl && tbaEl.textContent.trim() === trackingId) {
            return {
              found: true, row,
              date: this.extractField(row, 'date'),
              reason: this.extractField(row, 'Reason'),
              defect: this.extractField(row, 'Defect'),
              placeId: this.extractField(row, 'Place Id'),
            };
          }
        }
        return { found: false };
      },
      extractField(row, fieldName) {
        if (fieldName === 'date') {
          const dateEl = row.querySelector('.css-j4z1in ' + DOM_SELECTORS.PAST_DEL_DATE);
          return dateEl ? dateEl.textContent.trim() : 'N/A';
        }
        const labels = row.querySelectorAll(DOM_SELECTORS.PAST_DEL_TBA);
        for (const label of labels) {
          const text = label.textContent.trim();
          if (text === fieldName + ' |' || text === fieldName + ' | ') {
            const valueEl = label.nextElementSibling;
            if (valueEl) return valueEl.textContent.trim();
          }
        }
        return null;
      },
      clearHighlights() {
        document.querySelectorAll('.pbg-delivery-highlight').forEach(el => el.classList.remove('pbg-delivery-highlight'));
      },
    };

    const CaseDetailsPanel = {

      create() {
        if (this.element) return this.element;

      positionPanel() {
        if (!this.element) return;
        if (this._userMoved) return; // user dragged it — don't reset position
        const layersContainer = document.querySelector(DOM_SELECTORS.LAYERS_CONTAINER);
        const layersBtn = document.querySelector(DOM_SELECTORS.LAYERS_BUTTON);
        if (layersContainer && layersBtn) {
          const rect = layersContainer.getBoundingClientRect();
          const btnRect = layersBtn.getBoundingClientRect();
          this.element.style.position = 'fixed';
          this.element.style.left = rect.left + 'px';
          this.element.style.top = (btnRect.bottom + 8) + 'px';
        } else {
          this.element.style.position = 'fixed';
          this.element.style.left = '16px';
          this.element.style.top = '120px';
        }
      },

      setupModalObserver() {
        if (this.modalObserver) return;
        this.modalObserver = new MutationObserver(() => {
          const submitModal = document.querySelector(DOM_SELECTORS.SUBMIT_MODAL);
          const isViewingCase = document.querySelector(DOM_SELECTORS.LIVE_CASE_INDICATOR);
          if (submitModal) {
            if (!this.isSubmissionInProgress) { this.isSubmissionInProgress = true; this.hide(); }

    const UnifiedObserver = {
      observer: null, debounceTimers: new Map(),
      callbacks: { onAddressChange: [], onSelectionChange: [], onGeofenceChange: [] },
      start() {
        if (this.observer) return;
        this.observer = new MutationObserver((mutations) => this.handleMutations(mutations));
        this.observer.observe(document.body, { childList: true, subtree: true, attributes: true, attributeFilter: ["aria-checked", "checked", "value"] });
      },
      handleMutations(mutations) {
        const c = { addressChanged: false, selectionChanged: false, geofenceChanged: false };
        for (const m of mutations) {
          if (m.target.textContent?.includes("Address ID:")) c.addressChanged = true;
          if (m.type === "attributes" && (m.attributeName === "aria-checked" || m.attributeName === "checked")) c.selectionChanged = true;
          if (m.target.id === "geofence" || m.target.id === "poly_geofence") c.geofenceChanged = true;
        }
        if (c.addressChanged) this.debounce("address", () => this.triggerCallbacks("onAddressChange"), CONFIG.DEBOUNCE_DELAY);
        if (c.selectionChanged) this.debounce("selection", () => this.triggerCallbacks("onSelectionChange"), CONFIG.SELECTION_THROTTLE);
        if (c.geofenceChanged) this.debounce("geofence", () => this.triggerCallbacks("onGeofenceChange"), 300);
      },
      debounce(key, cb, d) {
        if (this.debounceTimers.has(key)) clearTimeout(this.debounceTimers.get(key));
        this.debounceTimers.set(key, setTimeout(() => { cb(); this.debounceTimers.delete(key); }, d));
      },
      on(event, cb) { if (this.callbacks[event]) this.callbacks[event].push(cb); },
      triggerCallbacks(event) { if (this.callbacks[event]) this.callbacks[event].forEach(cb => cb()); },
      stop() { if (this.observer) { this.observer.disconnect(); this.observer = null; } },
    };

          const isValid = curTT && curTT !== "EXTRACTION_FAILED" && curTT !== "N/A";
          const changed = isValid && this.lastTaskTemplateId !== null && curTT !== this.lastTaskTemplateId;
          const isFirst = isValid && this.lastTaskTemplateId === null;
          const prevDec = this.lastTaskTemplateId ? this.lastTaskTemplateId + (this.lastPbgEnabled === "true" ? "_PID" : "") : null;
          const curDec = isValid ? curTT + (pbg === "true" ? "_PID" : "") : null;

          CaseDetailsPanel.update(CaseDetailsManager.caseData, changed, prevDec);
          if (isFirst || changed) CaseDetailsPanel.showQueueChangeNotification(curDec, changed ? prevDec : null);
          if (isValid) { this.lastTaskTemplateId = curTT; this.lastPbgEnabled = pbg; }

          this.triggerPastDeliveries();
          await this.checkQueueAlerts();
          const gamPassed = await this.validateGAM();
          if (!gamPassed) return;
          if (CaseDetailsManager.getCaseField("pbgEnabled") !== "true") return;
          await this.extractInitialData();
          const result = await this.classify();
          this.startMonitoring(result);
        } catch (e) {
          NotificationManager.show("ERROR", "Processing Error", "Failed to process case. Please refresh and try again.", CONFIG.NOTIFICATION_DURATION.ERROR);
        }
      },

      async extractCaseDetails() {
        if (this.processingState.caseDetailsExtracted) return;
        await CaseDetailsManager.extractCaseDetails(this.currentAddressId);
        this.processingState.caseDetailsExtracted = true;
      },

      triggerPastDeliveries() {
        if (this.processingState.pastDeliveriesOpened) return;
        this.processingState.pastDeliveriesOpened = true;
        PastDeliveriesHelper.autoOpenAndSetFilters();
      },

      async checkQueueAlerts() {
        if (this.processingState.queueAlertShown) return;
        const tt = CaseDetailsManager.getCaseField("TaskTemplateId");
        const sgDP = CaseDetailsManager.getCaseField("SuggestedGeocodesDP");
        if (QueueDetector.shouldShowSuggestedGeocodesAlert(tt, sgDP)) { NotificationManager.showQueueAlert("SUGGESTED_GEOCODES", tt); this.processingState.queueAlertShown = true; return; }
        if (QueueDetector.shouldShowPOMAlert(tt)) { NotificationManager.showQueueAlert("POM_GEOCODE", tt); this.processingState.queueAlertShown = true; return; }
        this.processingState.queueAlertShown = true;
      },

      async validateGAM() {
        if (this.processingState.gamValidated) return true;
        const ad = DOMExtractor.extractAddressData();
        if (!ad.unitAddress || !ad.customerAddress) { NotificationManager.show("ERROR", "GAM Validation", "Failed to extract addresses", CONFIG.NOTIFICATION_DURATION.GAM); return false; }
        const comp = GAMValidator.validate(ad.unitAddress, ad.customerAddress);
        if (comp.isMatch) {
          NotificationManager.show("SUCCESS", "GAM Validation", "No GAM Issue - Addresses match correctly", CONFIG.NOTIFICATION_DURATION.GAM);
          this.processingState.gamValidated = true;
          await new Promise(r => setTimeout(r, 4500));
          return true;
        }
        const ci = { taskTemplateId: CaseDetailsManager.getCaseField("TaskTemplateId") || "N/A", pbgEnabled: CaseDetailsManager.getCaseField("pbgEnabled") || "N/A" };
        const action = await NotificationManager.showGAMModal(ci);
        if (action === "proceed") { this.processingState.gamValidated = true; return true; }
        NotificationManager.show("ERROR", "GAM Validation", "GAM Issue Confirmed - Please correct before proceeding", CONFIG.NOTIFICATION_DURATION.GAM);
        return false;
      },

      async extractInitialData() {
        if (this.processingState.initialDataExtracted) return;
        CaseDetailsManager.extractInitialAddressData();
        this.processingState.initialDataExtracted = true;
      },

      async classify() {
        const { pdl, dh } = CaseDetailsManager.getInitialAddressData();
        if (pdl === "EXTRACTION_FAILED" || dh === "EXTRACTION_FAILED") { NotificationManager.show("ERROR", "Classification Failed", "Unable to extract PDL or DH data. Please verify manually.", CONFIG.NOTIFICATION_DURATION.ERROR); return null; }
        const result = ClassificationValidator.classify(pdl, dh);
        if (result.needsVerification) {
          const action = await NotificationManager.showClassificationModal(result.reason);
          result.classification = action === "unit" ? "UNIT LEVEL" : "ADDRESS LEVEL";
          result.needsVerification = false;
          result.reason = "User verified as " + result.classification;
        }
        NotificationManager.show(
          result.classification === "UNIT LEVEL" ? "SUCCESS" : "WARNING", "Classification",
          "Case " + this.currentAddressId + "\nClassification: " + result.classification + "\nReason: " + result.reason,
          CONFIG.NOTIFICATION_DURATION.CLASSIFIER
        );
        this.processingState.classified = true;
        return result;
      },

      startMonitoring(result) { if (!result) return; SelectionMonitor.start(result); },

      reset(addressId) {
        this.currentAddressId = addressId;
        this.processingState = {
          caseDetailsExtracted: false, initialDataExtracted: false,
          gamValidated: false, classified: false, queueAlertShown: false,
          pastDeliveriesOpened: false,
        };
        CaseDetailsManager.clear(); DOMExtractor.clearCache();
        NotificationManager.clearAll(); SelectionMonitor.remove();
        PastDeliveriesHelper.clearHighlights();
        CaseDetailsPanel._userMoved = false; // new case → reset to default position

    function calcDist(lat1, lon1, lat2, lon2) {
      const R = 6371000;
      const a = Math.sin((lat2-lat1)*Math.PI/180/2)**2 + Math.cos(lat1*Math.PI/180)*Math.cos(lat2*Math.PI/180)*Math.sin((lon2-lon1)*Math.PI/180/2)**2;
      return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
    }

    function parseCoordinates(value) {
      if (!value || typeof value !== 'string') return null;
      const trimmed = value.trim();
      if (!trimmed || !trimmed.includes(',')) return null;
      try {
        const [lat, lon] = trimmed.split(',').map(c => parseFloat(c.trim()));
        if (isNaN(lat) || isNaN(lon)) return null;
        if (lat < -90 || lat > 90 || lon < -180 || lon > 180) return null;
        if (lat === 0 && lon === 0) return null;
        return { lat, lon, original: trimmed };
      } catch { return null; }
    }

      if (document.getElementById('geo-suggested-dp-container')) return document.getElementById('geo-suggested-dp-input');
      const transferSection = document.querySelector('[data-testid="transfer-form-checkbox"]')?.closest('.css-1c02o5u') ||
                              document.querySelector('[data-testid="transfer-form-checkbox"]')?.closest('.css-1d7jqjm')?.parentElement;
      if (!transferSection) return null;
      const container = document.createElement('div');
      container.id = 'geo-suggested-dp-container';
      container.innerHTML = `
        <div style="margin-top:12px;padding:10px 14px;background:#f8f9fa;border:1px solid #dee2e6;border-radius:6px;">
          <label style="display:block;font-weight:600;font-size:13px;margin-bottom:6px;color:#495057;">
            Suggested DP Coordinates <span style="color:#d32f2f;">*</span>
          </label>
          <input id="geo-suggested-dp-input" type="text" placeholder="lat, lon (e.g. 37.7749, -122.4194)"
            style="width:100%;padding:8px 10px;border:2px solid #dee2e6;border-radius:4px;font-size:13px;box-sizing:border-box;outline:none;" />
          <div id="geo-dp-hint" style="font-size:12px;color:#6c757d;margin-top:4px;">Must be ≥10km from current DP geocode.</div>
        </div>`;
      transferSection.appendChild(container);
      return document.getElementById('geo-suggested-dp-input');
    }

    function removeFallbackDPField() { document.getElementById('geo-suggested-dp-container')?.remove(); }

    function getDPFieldValue() {
      const native = getNativeDPField();
      if (native?.value) return native.value.trim();
      const fallback = document.getElementById('geo-suggested-dp-input');
      return fallback?.value?.trim() || null;
    }

    function validateDPField() {
      const dpValue = getDPFieldValue();
      if (!dpValue) return { valid: false, error: 'No DP coordinate entered.' };
      const coords = parseCoordinates(dpValue);
      if (!coords) return { valid: false, error: 'Invalid coordinate format. Use: lat, lon' };
      if (!storedDPGeocode) return { valid: true, coords };
      const distance = calcDist(storedDPGeocode.lat, storedDPGeocode.lon, coords.lat, coords.lon);
      const distanceKm = (distance / 1000).toFixed(2);
      if (distance < 10000) return { valid: false, error: `Distance too small: ${distanceKm}km. Must be ≥10km from current DP.` };
      const hint = document.getElementById('geo-dp-hint');
      if (hint) { hint.style.color = '#4caf50'; hint.textContent = `Valid! Distance: ${distanceKm}km from current DP`; }
      return { valid: true, coords, distance };
    }

    function filterQueueOptions() {
      const optionsList = document.querySelector('[role="listbox"]') || document.querySelector('[id*="options-list"]');
      if (!optionsList) return;
      const gamYes = isGamIssueYes();
      optionsList.querySelectorAll('[role="option"]').forEach(option => {
        const t = option.textContent.trim();
        // Only disable DPRE queues when GAM Issue = Yes (PBG blocker removed per user request)
        const shouldDisable = gamYes && isDpreQueue(t);
        option.style.opacity = shouldDisable ? '0.4' : '1';
        option.style.pointerEvents = shouldDisable ? 'none' : 'auto';
        option.style.cursor = shouldDisable ? 'not-allowed' : 'pointer';
        option.style.backgroundColor = shouldDisable ? '#f0f0f0' : '';
        option.style.textDecoration = shouldDisable ? 'line-through' : 'none';
      });
    }

    function validateQueueSelection(selectedQueue) {
      if (!selectedQueue) return true;
      if (transferLocked && isPbgQueue(selectedQueue)) return true;
      if (upidTransferLocked && selectedQueue === UPID_QUEUE) return true;
      // Only block DPRE queues when GAM Issue = Yes
      if (isGamIssueYes() && isDpreQueue(selectedQueue)) {
        showGAMPopup('error','Queue Blocked','<strong>DPRE queues</strong> are not allowed when <strong>GAM Issue = Yes</strong>. Please select a PBG queue instead.');
        return false;
      }
      // Removed: PBG blocker when GAM=No (per user request)
      // Removed: distance deviation check for Suggested DP (per user request)
      return true;
    }

    function handleQueueSelection(selectedQueue) {
      if (currentSelectedQueue === selectedQueue) return;
      currentSelectedQueue = selectedQueue;
      if (!isDpreQueue(selectedQueue) || isGamIssueYes()) removeFallbackDPField();
      const isValid = validateQueueSelection(selectedQueue);
      if (!isValid) currentSelectedQueue = null;
    }

    function handleGamIssueChange(newValue) {
      if (newValue === 'Yes') {
        // GAM=Yes: highest priority. Kill UPID lock, take over with PBG.
        upidTransferLocked = false;
        removeFallbackDPField();
        setTimeout(() => autoSelectPbgTransfer(), 100);
      } else {
        // GAM changed to No/NA: release GAM lock.
        transferLocked = false;
        // Check UPID — if UPID=No, hand off to UPID queue. Otherwise clear transfer.
        const upidVal = getUpidGeocodeValue();
        if (upidVal && upidVal.toLowerCase() === 'no') {
          // UPID=No takes over now
          setTimeout(() => autoSelectUpidQueue(), 100);
        } else {
          // Nothing active — uncheck transfer if it was on a PBG queue
          upidTransferLocked = false;
          applyLockUI();
          const cq = getCurrentSelectedQueue();
          if (cq && isPbgQueue(cq)) { disableGAMTransfer(); currentSelectedQueue = null; }
        }
      }
      setTimeout(broadcastTransferState, 800);
    }

    function setupGamIssueListenersGAM() {
      const sec = getGamIssueSection(); if (!sec) return;
      sec.querySelectorAll('input[type="radio"]').forEach(radio => {
        if (radio._geoListener) return; radio._geoListener = true;
        radio.addEventListener('change', () => {
          if (radio.checked) {
            currentGamIssueValue = radio.value;
            handleGamIssueChange(radio.value);
            setTimeout(filterQueueOptions, 100);
            if (!transferLocked) {
              const cq = getCurrentSelectedQueue();
              if (cq) { const isValid = validateQueueSelection(cq); if (!isValid) disableGAMTransfer(); }
            }
          }
        });
      });
    }

    function setupQueueSelectorListenersGAM() {
      const qs = document.querySelector('[data-testid="transfer-form-queue-selector"]');
      if (!qs || qs._geoListener) return; qs._geoListener = true;
      qs.addEventListener('click', (e) => {
        if (transferLocked && !isScriptClicking) { e.preventDefault(); e.stopPropagation(); showGAMPopup('warning','Transfer Locked','Transfer is locked because GAM Issue = Yes. Change GAM Issue to No to modify transfer settings.'); return; }
        setTimeout(filterQueueOptions, 50); setTimeout(filterQueueOptions, 150);
      });
      new MutationObserver(() => {
        if (isScriptClicking) return;
        const cq = getCurrentSelectedQueue();
        if (cq && cq !== currentSelectedQueue) handleQueueSelection(cq);
      }).observe(qs, { childList: true, subtree: true, characterData: true });
    }

    function setupTransferCheckboxListenerGAM() {
      const cb = document.querySelector('[data-testid="transfer-form-checkbox"]');
      if (!cb || cb._geoListener) return; cb._geoListener = true;
      cb.addEventListener('click', (e) => {
        if (transferLocked && !isScriptClicking) {
          e.preventDefault(); e.stopPropagation();
          showGAMPopup('warning','Transfer Locked','Transfer is locked because GAM Issue = Yes. Change GAM Issue to No to modify transfer settings.');
          return;
        }
        setTimeout(() => {
          if (isTransferEnabled() && !transferLocked) {
            const cq = getCurrentSelectedQueue();
            if (cq) { const isValid = validateQueueSelection(cq); if (!isValid) disableGAMTransfer(); }
          }
        }, 100);
      }, true);
    }

    function setupOptionClickListenerGAM() {
      document.addEventListener('click', (e) => {
        if (isScriptClicking) return;
        const option = e.target.closest('[role="option"]');
        if (!option) return;
        const optionText = option.textContent.trim();
        if (isDpreQueue(optionText) || isPbgQueue(optionText)) {
          setTimeout(() => { const isValid = validateQueueSelection(optionText); if (!isValid) disableGAMTransfer(); }, 100);
        }
      }, true);
    }

    function setupIframeCommunication() {
      window.addEventListener('message', (event) => {
        if (event.data?.type === 'GEOSTUDIO_DP_GEOCODE') storedDPGeocode = event.data.data;
      });
      function requestDPGeocode() {
        try { window.parent.postMessage({ type: 'REQUEST_DP_GEOCODE' }, '*'); } catch (e) {}
      }
      setTimeout(requestDPGeocode, 500);
      setInterval(requestDPGeocode, 3000);
    }

    // Broadcast transfer-required state to parent (for submit button grey-out)
    function broadcastTransferState() {
      const gamYes  = isGamIssueYes();
      const upidNo  = isUpidNo();
      const needsTransfer = gamYes || upidNo;
      const hasTransfer   = isTransferEnabled() && !!getCurrentSelectedQueue();
      const required = needsTransfer && !hasTransfer;
      let reason = '';
      if (required) {
        if (gamYes && upidNo)   reason = '⚠ GAM Issue = Yes & UPID = No: select transfer task before submitting';
        else if (gamYes)        reason = '⚠ GAM Issue = Yes: select PBG transfer task before submitting';
        else if (upidNo)        reason = '⚠ UPID Correction = No: select UPID transfer task before submitting';
      }
      try { window.parent.postMessage({ type: 'GEOSTUDIO_TRANSFER_REQUIRED', required, reason }, '*'); } catch (e) {}
    }

    function checkInitialGamState() {
      const val = getCurrentGamIssueValue(); currentGamIssueValue = val;
      if (val === 'Yes') setTimeout(() => autoSelectPbgTransfer(), 500);
    }

    function applyInitialQueueState() {
      const gamVal  = getCurrentGamIssueValue();
      const upidVal = getUpidGeocodeValue();
      if (gamVal === 'Yes') {
        upidTransferLocked = false;
        autoSelectPbgTransfer();
        return;
      }
      if (upidVal && upidVal.toLowerCase() === 'no') {
        transferLocked = false;
        autoSelectUpidQueue();
        return;
      }
      transferLocked = false;
      upidTransferLocked = false;
      applyLockUI();
      broadcastTransferState();
    }

    // Pre-warm the transfer queue dropdown so options are loaded into DOM
    // before any auto-select fires. The queue selector only renders its option
    // list after the transfer checkbox is enabled AND the dropdown is opened at
    // least once. We simulate this silently on load.
    function closeParentDialog() {
      const findAndClickClose = () => {
        const dialog = document.querySelector('div[role="dialog"][aria-modal="true"]');
        if (dialog) {
          const btn = dialog.querySelector('button[aria-label="Close"]') || dialog.querySelector('button.css-148awne');
          if (btn) { btn.click(); return true; }
        }
        return false;
      };
      if (!findAndClickClose()) setTimeout(findAndClickClose, 200);
    }

    window.addEventListener('message', (event) => {
      if (event.data?.type === 'GEOSTUDIO_CLOSE_DIALOG') closeParentDialog();
    });

    // ── UPID Geocode Accuracy → Auto-select RDR_COMBINED_UPID_TRIAGE ──────────
    // When "Upid Geocode Accuracy" field has "Yes" selected (css-1prjt9m div)
    // automatically enable transfer and select RDR_COMBINED_UPID_TRIAGE queue.
    const UPID_QUEUE = 'RDR_COMBINED_UPID_TRIAGE';
    let upidTransferLocked = false;

    function getUpidGeocodeSection() {
      const candidates = document.querySelectorAll('p.css-17xh5uu');
      for (const lbl of candidates) {
        const normalized = lbl.textContent.trim().toLowerCase().replace(/\s+/g, '');
        if (normalized === 'upidgeocodeaccuracy') {
          return lbl.closest('.css-19hedjk') || lbl.closest('div') || lbl.parentElement;
        }
      }
      return null;
    }

    function getUpidGeocodeValue() {
      const sec = getUpidGeocodeSection();
      if (!sec) return null;
      // Primary: selected div has class css-1prjt9m
      const selected = sec.querySelector('div.css-1prjt9m');
      if (selected) return selected.textContent.trim();
      // Fallback: checked radio input
      for (const r of sec.querySelectorAll('input[type="radio"]')) {
        if (r.checked) return r.value.trim();
      }
      // Fallback: aria-checked
      for (const el of sec.querySelectorAll('[role="radio"]')) {
        if (el.getAttribute('aria-checked') === 'true') return el.textContent.trim();
      }
      return null;
    }

    function isUpidNo() {
      const v = getUpidGeocodeValue();
      return v && v.toLowerCase() === 'no';
    }

    function applyUpidTransferLock() { applyLockUI(); } // legacy alias — state already set

    function removeUpidTransferLock() {
      upidTransferLocked = false;
      applyLockUI();
    }

    function autoSelectUpidQueue() {
      // Set intent flags immediately
      transferLocked = false;
      upidTransferLocked = true;
      // Remove GAM lock indicator right away, but DON'T apply UPID lock yet —
      // we need the selector to be fully interactive for our script clicks.
      document.getElementById('geo-transfer-lock-indicator')?.remove();
      document.getElementById('upid-transfer-lock-indicator')?.remove();
      const cb = document.querySelector('[data-testid="transfer-form-checkbox"]');
      const qs = document.querySelector('[data-testid="transfer-form-queue-selector"]');
      // Fully unlock UI for our clicks
      if (cb) { cb.style.opacity = '1'; cb.style.pointerEvents = 'auto'; cb.style.cursor = 'pointer'; }
      if (qs) { qs.style.opacity = '1'; qs.style.pointerEvents = 'auto'; qs.style.cursor = 'pointer'; }

      enableGAMTransfer();
      setTimeout(() => {
        const selector = document.querySelector('[data-testid="transfer-form-queue-selector"]');
        if (!selector) { applyLockUI(); return; }
        isScriptClicking = true;
        selector.dispatchEvent(new MouseEvent('mousedown', { bubbles: true, cancelable: true, view: window }));
        selector.dispatchEvent(new MouseEvent('click',     { bubbles: true, cancelable: true, view: window }));
        isScriptClicking = false;
        setTimeout(() => {
          const options = document.querySelectorAll('[role="option"]');
          let found = false;
          for (const opt of options) {
            if (opt.textContent.trim() === UPID_QUEUE) {
              isScriptClicking = true; opt.click(); isScriptClicking = false;
              currentSelectedQueue = UPID_QUEUE; found = true;
              break;
            }
          }
          // NOW apply the visual lock — queue is already changed
          applyLockUI();
          if (found) showGAMPopup('success', 'UPID Queue Auto-Selected',
            `<strong>${UPID_QUEUE}</strong> auto-selected because <strong>Upid Geocode Accuracy = No</strong>.`);
          setTimeout(broadcastTransferState, 300);
        }, 500);
      }, 300);
    }

    function handleUpidChange(value) {
      // GAM=Yes always has priority — UPID must not override it
      if (isGamIssueYes()) {
        upidTransferLocked = false;
        // Don't touch transferLocked or the UI — GAM is in control
        return;
      }
      if (value && value.toLowerCase() === 'no') {
        // UPID=No: take over with UPID queue
        setTimeout(() => autoSelectUpidQueue(), 100);
      } else {
        // UPID changed away from No: release UPID lock, clear transfer if it was UPID queue
        upidTransferLocked = false;
        applyLockUI();
        const cq = getCurrentSelectedQueue();
        if (cq === UPID_QUEUE) { disableGAMTransfer(); currentSelectedQueue = null; }
      }
      setTimeout(broadcastTransferState, 800);
    }

    function setupUpidGeocodeListener() {
      const sec = getUpidGeocodeSection();
      if (!sec || sec._upidListener) return;
      sec._upidListener = true;

      // Watch for radio input change
      sec.querySelectorAll('input[type="radio"]').forEach(r => {
        if (r._upidBound) return; r._upidBound = true;
        r.addEventListener('change', () => { if (r.checked) handleUpidChange(r.value); });
      });

      // Single click listener on whole section — covers div-based option UIs
      sec.addEventListener('click', () => {
        setTimeout(() => handleUpidChange(getUpidGeocodeValue() || ''), 100);
      });

      // MutationObserver: catches React class-toggle selection changes
      new MutationObserver(() => {
        if (isGamIssueYes()) return; // GAM = Yes always has priority — don't react to UPID changes
        if (isUpidNo() && !upidTransferLocked) handleUpidChange('no');
        else if (!isUpidNo() && upidTransferLocked) handleUpidChange('other');
      }).observe(sec, { attributes: true, subtree: true, attributeFilter: ['class', 'aria-checked'] });
    }

    // ── Initialize Iframe features ─────────────────────────────────────────────
    (async function initIframe() {
      await delay(600);

      // NEI button (non-blocking)
      injectButtonsNearGeocode();

      // GAM Issue auto-checkbox
      setupGAMIssueAutoTransfer();

      // GAM DP Triage controller
      setupIframeCommunication();
      setupGamIssueListenersGAM();
      setupQueueSelectorListenersGAM();
      setupOptionClickListenerGAM();
      setupTransferCheckboxListenerGAM();

      // smartPreWarm: waits for transfer form to appear, loads queue options, applies state
      (async function smartPreWarm() {
        for (let attempt = 0; attempt < 25; attempt++) {
          const checkbox = document.querySelector('[data-testid="transfer-form-checkbox"]');
          const selector = document.querySelector('[data-testid="transfer-form-queue-selector"]');
          if (checkbox && selector) {
            const wasEnabled = checkbox.getAttribute('aria-checked') === 'true';
            if (!wasEnabled) { isScriptClicking = true; checkbox.click(); isScriptClicking = false; await delay(300); }
            selector.dispatchEvent(new MouseEvent('mousedown', { bubbles: true }));
            await delay(400);
            document.body.click();
            await delay(150);
            if (!wasEnabled && !transferLocked && !upidTransferLocked) {
              isScriptClicking = true; checkbox.click(); isScriptClicking = false;
            }
            setupUpidGeocodeListener();
            applyInitialQueueState();
            return;
          }
          await delay(400);
        }
      })();

      // Periodic re-setup for dynamic DOM
      setInterval(() => {
        setupGamIssueListenersGAM();
        setupQueueSelectorListenersGAM();
        setupTransferCheckboxListenerGAM();
        setupUpidGeocodeListener();
        broadcastTransferState();
        if (!transferLocked) {
          const cq = getCurrentSelectedQueue();
          if (cq && cq !== currentSelectedQueue) handleQueueSelection(cq);
        }
      }, 2000);

      // MutationObserver — re-attaches listeners when new form elements appear
      new MutationObserver((mutations) => {
        let shouldSetup = false;
        for (const m of mutations) {
          if (m.type === 'childList') {
            for (const node of m.addedNodes) {
              if (node.nodeType === Node.ELEMENT_NODE &&
                  (node.querySelector?.('[data-testid="transfer-form-queue-selector"]') ||
                   node.querySelector?.('.css-17xh5uu') ||
                   node.matches?.('[role="listbox"]'))) {
                shouldSetup = true; break;
              }
            }
          }
          if (shouldSetup) break;
        }
        if (shouldSetup) {
          setupGamIssueListenersGAM();
          setupQueueSelectorListenersGAM();
          setupTransferCheckboxListenerGAM();
          setupUpidGeocodeListener();
          setTimeout(filterQueueOptions, 100);
        }
      }).observe(document.body, { childList: true, subtree: true });

      console.log('GeoStudio Combined V12.1 — iframe initialized');
    })();

  } // end IS_IFRAME

  // ── Main page: broadcast DP geocode to iframes (from GAMAutoFill) ─────────────
  if (IS_MAIN) {
    let _currentAddressId = null;
    let _currentDPGeocode = null;

    function _getAddressId() {
      try {
        const el = document.evaluate("//p[text()='Address ID:']/following-sibling::a/p", document, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;
        return el ? el.textContent.trim() : null;
      } catch { return null; }
    }

    function _parseCoordinates(value) {
      if (!value || typeof value !== 'string' || !value.includes(',')) return null;
      try {
        const [lat, lon] = value.split(',').map(c => parseFloat(c.trim()));
        if (isNaN(lat) || isNaN(lon) || lat < -90 || lat > 90 || lon < -180 || lon > 180 || (lat === 0 && lon === 0)) return null;
        return { lat, lon, original: value.trim() };
      } catch { return null; }
    }

    function _broadcastDPToIframes(coords) {
      document.querySelectorAll('iframe').forEach(iframe => {
        try { iframe.contentWindow.postMessage({ type: 'GEOSTUDIO_DP_GEOCODE', data: coords }, '*'); } catch (e) {}
      });
    }

    function _captureInitialDPGeocode(addressId) {
      if (!addressId) return;
      const dpEl = document.querySelector('input#input-dp-geocode');
      if (!dpEl || !dpEl.value) { setTimeout(() => _captureInitialDPGeocode(addressId), 1000); return; }
      const coords = _parseCoordinates(dpEl.value);
      if (coords) { _currentDPGeocode = coords; _broadcastDPToIframes(coords); }
    }

    window.addEventListener('message', (event) => {
      if (event.data?.type === 'REQUEST_DP_GEOCODE') {
        if (_currentDPGeocode) event.source.postMessage({ type: 'GEOSTUDIO_DP_GEOCODE', data: _currentDPGeocode }, '*');
      }
      // Submit grey-out: iframe sends transfer requirement state
      if (event.data?.type === 'GEOSTUDIO_TRANSFER_REQUIRED') {
        _handleTransferRequiredState(event.data.required, event.data.reason);
      }
    });

    let _transferRequired = false;
    let _transferReason = '';
    let _submitStyle = null;

    function _getSubmitBtn() {
      return document.querySelector('#submit-btn') ||
             document.querySelector('button[aria-controls="submit"]');
    }

    function _handleTransferRequiredState(required, reason) {
      // Normalise reason — never let it be undefined/null
      const newReason = reason || _transferReason || 'Select a transfer task before submitting';
      // Only update the banner text if something actually changed
      const stateChanged  = required !== _transferRequired;
      const reasonChanged = newReason !== _transferReason;
      _transferRequired = required;
      if (required) _transferReason = newReason;

      const btn = _getSubmitBtn();
      if (!btn) return;
      if (required) {
        if (!_submitStyle) {
          _submitStyle = { opacity: btn.style.opacity, cursor: btn.style.cursor, pointerEvents: btn.style.pointerEvents, title: btn.title };
        }
        btn.style.opacity = '0.4';
        btn.style.cursor = 'not-allowed';
        btn.style.pointerEvents = 'none';
        btn.title = newReason;

        const existing = document.getElementById('pbg-submit-warning');
        if (!existing) {
          const w = document.createElement('div');
          w.id = 'pbg-submit-warning';
          w.style.cssText = 'position:fixed;top:20px;right:20px;background:#d32f2f;color:#fff;padding:16px;border-radius:8px;font-size:14px;font-weight:600;z-index:9999;box-shadow:0 5px 15px rgba(0,0,0,0.15);max-width:380px;line-height:1.5;display:flex;align-items:flex-start;gap:12px;border-left:5px solid #b71c1c;';
          w.innerHTML = `<div style="flex-shrink:0;margin-top:2px;"><svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="#fff" stroke-width="2"><path d="M10.29 3.86L1.82 18a2 2 0 001.71 3h16.94a2 2 0 001.71-3L13.71 3.86a2 2 0 00-3.42 0z"/><line x1="12" y1="9" x2="12" y2="13"/><line x1="12" y1="17" x2="12.01" y2="17"/></svg></div><div style="flex-grow:1;"><div style="font-size:16px;font-weight:700;margin-bottom:4px;">Transfer Required</div><div id="pbg-submit-warning-msg" style="font-size:13px;font-weight:400;opacity:0.95;">${newReason}</div></div>`;
          document.body.appendChild(w);
        } else if (reasonChanged) {
          // Only update the message text when the reason actually changes
          const msgEl = document.getElementById('pbg-submit-warning-msg');
          if (msgEl) msgEl.textContent = newReason;
        }
      } else {
        _transferReason = '';
        if (_submitStyle) {
          btn.style.opacity      = _submitStyle.opacity      || '';
          btn.style.cursor       = _submitStyle.cursor       || '';
          btn.style.pointerEvents = _submitStyle.pointerEvents || '';
          btn.title              = _submitStyle.title        || '';
          _submitStyle = null;
        }
        document.getElementById('pbg-submit-warning')?.remove();
      }
    }

    // Poll for submit button existence and re-apply state when it appears
    setInterval(() => {
      if (_transferRequired) _handleTransferRequiredState(true, _transferReason);
    }, 1000);

    new MutationObserver(() => {
      const newId = _getAddressId();
      if (newId && newId !== _currentAddressId) {
        _currentAddressId = newId; _currentDPGeocode = null;
        setTimeout(() => _captureInitialDPGeocode(newId), 500);
